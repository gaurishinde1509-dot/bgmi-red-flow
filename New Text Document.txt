    <!-- JavaScript to Handle Step Switching and UPI Generation -->
    <script>
        // Transition to next page
        function nextStep(event, nextStepNum) {
            event.preventDefault(); // Stop form submission reloading
            goToStep(nextStepNum);

            // If moving to step 3 (Payment), generate the dynamic QR
            if (nextStepNum === 3) {
                generatePaymentQR();
            }
        }

        // Switch visible wrapper
        function goToStep(stepNum) {
            // Hide all steps
            document.querySelectorAll('.step').forEach(step => {
                step.classList.remove('active');
            });
            // Show requested step
            document.getElementById(`step-${stepNum}`).classList.add('active');
        }

        // Generate customized UPI QR Code
        function generatePaymentQR() {
            const qrContainer = document.getElementById("qrcode");
            qrContainer.innerHTML = ""; // Clear existing QR code if any

            const upiId = "yourname@okaxis";  // ⚠️ Change to your UPI ID
            const payeeName = "Tournament Playz";
            const amount = "5000";
            const teamName = document.getElementById("team-name").value || "Team";
            const transactionNote = `Entry Fee for ${teamName}`;

            // NPCI Official UPI Deep Link
            const upiLink = `upi://pay?pa=${encodeURIComponent(upiId)}&pn=${encodeURIComponent(payeeName)}&am=${amount}&cu=INR&tn=${encodeURIComponent(transactionNote)}`;

            // Build QR code inside placeholder
            new QRCode(qrContainer, {
                text: upiLink,
                width: 200,
                height: 200,
                colorDark : "#ffffff", // Light QR code color on dark background
                colorLight : "#1e293b", // Matching card-bg
                correctLevel : QRCode.CorrectLevel.H
            });
        }

        // Reset to Login (Page 1)
        function restartRegistration() {
            document.getElementById("login-id").value = "";
            document.getElementById("login-password").value = "";
            document.getElementById("team-name").value = "";
            document.getElementById("players").value = "";
            document.getElementById("whatsapp").value = "";
            goToStep(1);
        }
    </script>
